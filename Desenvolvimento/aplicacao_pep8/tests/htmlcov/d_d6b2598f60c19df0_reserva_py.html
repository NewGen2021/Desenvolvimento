24def converteNumeroDiaSemana(weekday, To_ISO=True): 

25 # ISO-8601 [0 - monday, 1 - tuesday, 6 - sunday] 

26 # Não ISO [0 - sunday , 1 - monday , 6 - saturday] 

27 if To_ISO: 

28 if weekday == 0: 

29 return 6 

30 return weekday - 1 

31 else: 

32 if weekday == 6: 

33 return 0 

34 return weekday + 1
